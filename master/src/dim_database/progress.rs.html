10pub struct Progress {

12    pub delta: i64,

18impl Progress {
19    pub async fn set(
20        conn: &mut crate::Transaction<'_>,
21        delta: i64,
22        uid: UserID,
23        mid: i64,
24    ) -> Result<usize, DieselError> {
25        let timestamp = SystemTime::now()
26            .duration_since(SystemTime::UNIX_EPOCH)
27            .unwrap()
28            .as_secs() as i64;
29
30        Ok(sqlx::query!(
31            "INSERT OR REPLACE INTO progress (delta, media_id, user_id, populated)
32            VALUES ($1, $2, $3, $4)",
33            delta,
34            mid,
35            uid,
36            timestamp
37        )
38        .execute(conn.as_mut())
39        .await?
40        .rows_affected() as usize)
41    }
42
43    pub async fn get_for_media_user(
44        conn: &mut crate::Transaction<'_>,
45        uid: UserID,
46        mid: i64,
47    ) -> Result<Self, DieselError> {
48        Ok(sqlx::query_as!(
49            Progress,
50            r#"SELECT id, user_id as "user_id: UserID", delta, media_id, populated FROM progress
51            WHERE user_id = ?
52            AND media_id = ?"#,
53            uid,
54            mid
55        )
56        .fetch_optional(conn.as_mut())
57        .await?
58        .unwrap_or(Self {
59            id: Default::default(),
60            media_id: mid,
61            user_id: uid,
62            delta: Default::default(),
63            populated: Default::default(),
64        }))
65    }

81    pub async fn get_total_for_media(
82        conn: &mut crate::Transaction<'_>,
83        media: &Media,
84        uid: UserID,
85    ) -> Result<i64, DieselError> {
86        match media.media_type {
87            MediaType::Tv => Ok(Self::get_total_for_tv(conn, uid, media.id).await? as i64),
88            _ => Ok(Self::get_for_media_user(conn, uid, media.id)
89                .await
90                .map(|x| x.delta)?),
91        }
92    }

94    pub async fn get_progress_for_media(
95        conn: &mut crate::Transaction<'_>,
96        id: i64,
97        uid: UserID,
98    ) -> Result<(i64, i64), DieselError> {
99        #[derive(sqlx::FromRow)]
100        struct Record {
101            delta: i64,
102            duration: i64,
103        }
104
105        // FIXME: Use query_as macro instead of query_as function when https://github.com/launchbadge/sqlx/issues/1249 is fixed.
106        let record = sqlx::query_as::<_, Record>(
107            "SELECT progress.delta, MAX(mediafile.duration) duration FROM _tblmedia
108            INNER JOIN mediafile ON mediafile.media_id = _tblmedia.id
109            LEFT OUTER JOIN progress ON progress.media_id = _tblmedia.id AND progress.user_id = ?
110            WHERE _tblmedia.id = ?
111            GROUP BY _tblmedia.id
112            LIMIT 1",
113        )
114        .bind(uid)
115        .bind(id)
116        .fetch_one(conn.as_mut())
117        .await?;
118
119        Ok((record.delta, record.duration))
120    }

122    pub async fn get_total_for_tv(
123        conn: &mut crate::Transaction<'_>,
124        uid: UserID,
125        tv_id: i64,
126    ) -> Result<i32, DieselError> {
127        #[derive(sqlx::FromRow)]
128        struct Row {
129            total: i32,
130        }
131
132        Ok(sqlx::query_as::<_, Row>(
133            "SELECT COALESCE(SUM(progress.delta), 0) as total FROM _tblmedia
134            JOIN progress ON progress.media_id = _tblmedia.id
135            JOIN episode ON episode.id = _tblmedia.id
136            JOIN season on season.id = episode.seasonid
137            
138            WHERE season.tvshowid = ?
139            AND progress.user_id = ?",
140        )
141        .bind(tv_id)
142        .bind(uid)
143        .fetch_one(conn.as_mut())
144        .await?
145        .total)
146    }

148    pub async fn get_continue_watching(
149        conn: &mut crate::Transaction<'_>,
150        uid: UserID,
151        count: i64,
152    ) -> Result<Vec<i64>, DieselError> {
153        Ok(sqlx::query_scalar(
154            r#"SELECT _tblmedia.id  FROM _tblmedia
155
156            JOIN season on season.tvshowid = _tblmedia.id
157            JOIN episode on episode.seasonid = season.id
158            JOIN progress on progress.media_id = episode.id
159            JOIN library on library.id = _tblmedia.library_id
160
161            WHERE NOT progress.populated = 0
162            AND progress.user_id = ?
163            AND NOT library.hidden
164
165            GROUP BY _tblmedia.id
166            ORDER BY progress.populated DESC
167            LIMIT ?"#,
168        )
169        .bind(uid)
170        .bind(count)
171        .fetch_all(conn.as_mut())
172        .await?)
173    }
